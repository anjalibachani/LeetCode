1. Two Sum

class Solution {
    public int[] twoSum(int[] nums, int target)
    {
        for(int i=0;i<=nums.length;i++)
        {
            for(int j=i+1;j<=nums.length - 1;j++)
            {
                if (nums[i] + nums[j] == target)
                {
                    return new int[] {i, j};   
                }
                else
                {
                    continue;
                }
            }
        }
        throw new NullPointerException("no two solutions");
    }
}
